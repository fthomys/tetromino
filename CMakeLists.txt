cmake_minimum_required(VERSION 3.30)
project(Tetromino C)

set(CMAKE_C_STANDARD 11)

message(STATUS "üîç Looking for SDL2...")
find_package(SDL2 REQUIRED)


message(STATUS "üîç Looking for SDL2_mixer...")
find_package(SDL2_mixer REQUIRED)

find_package(SDL2_ttf REQUIRED)


include_directories(
        ${SDL2_INCLUDE_DIRS}
        ${SDL2_MIXER_INCLUDE_DIRS}
        ${SDL2_TTF_INCLUDE_DIRS}
)

message(STATUS "üõ†Ô∏è  Building executable Tetromino...")
add_executable(Tetromino source/main.c source/piece.c)

set(FONT_SOURCE "${CMAKE_CURRENT_SOURCE_DIR}/assets/Arial.ttf")
if(EXISTS "${FONT_SOURCE}")
    message(STATUS "üìÅ Copying font: Arial.ttf to binary directory...")
    file(COPY "${FONT_SOURCE}" DESTINATION ${CMAKE_CURRENT_BINARY_DIR})
else()
    message(WARNING "‚ö†Ô∏è Font Arial.ttf not found in assets/. Make sure the file exists!")
endif()
